182{

196inline void Blackboard::unset(const std::string& key)
197{
198 std::unique_lock storage_lock(storage_mutex_);
199
200 // check local storage
201 auto it = storage_.find(key);
202 if(it == storage_.end())
203 {
204 // No entry, nothing to do.
205 return;
206 }
207
208 storage_.erase(it);

211template <typename T>
212inline void Blackboard::set(const std::string& key, const T& value)
213{
214 if(StartWith(key, '@'))
215 {
216 rootBlackboard()->set(key.substr(1, key.size() - 1), value);
217 return;
218 }
219 std::unique_lock storage_lock(storage_mutex_);
220
221 // check local storage
222 auto it = storage_.find(key);
223 if(it == storage_.end())
224 {
225 // create a new entry
226 Any new_value(value);
227 storage_lock.unlock();
228 std::shared_ptr<Blackboard::Entry> entry;
229 // if a new generic port is created with a string, it's type should be AnyTypeAllowed
230 if constexpr(std::is_same_v<std::string, T>)
231 {
232 entry = createEntryImpl(key, PortInfo(PortDirection::INOUT));
233 }
234 else
235 {
236 PortInfo new_port(PortDirection::INOUT, new_value.type(),
237 GetAnyFromStringFunctor<T>());
238 entry = createEntryImpl(key, new_port);
239 }
240 storage_lock.lock();
241
242 entry->value = new_value;
243 entry->sequence_id++;
244 entry->stamp = std::chrono::steady_clock::now().time_since_epoch();
245 }
246 else
247 {
248 // this is not the first time we set this entry, we need to check
249 // if the type is the same or not.
250 Entry& entry = *it->second;
251 storage_lock.unlock();
252
253 std::scoped_lock scoped_lock(entry.entry_mutex);
254
255 Any& previous_any = entry.value;
256 Any new_value(value);
257
258 // special case: entry exists but it is not strongly typed... yet
259 if(!entry.info.isStronglyTyped())
260 {
261 // Use the new type to create a new entry that is strongly typed.
262 entry.info = TypeInfo::Create<T>();
263 entry.sequence_id++;
264 entry.stamp = std::chrono::steady_clock::now().time_since_epoch();
265 previous_any = std::move(new_value);
266 return;
267 }
268
269 std::type_index previous_type = entry.info.type();
270
271 // check type mismatch
272 if(previous_type != std::type_index(typeid(T)) && previous_type != new_value.type())
273 {
274 bool mismatching = true;
275 if(std::is_constructible<StringView, T>::value)
276 {
277 Any any_from_string = entry.info.parseString(value);
278 if(any_from_string.empty() == false)
279 {
280 mismatching = false;
281 new_value = std::move(any_from_string);
282 }
283 }
284 // check if we are doing a safe cast between numbers
285 // for instance, it is safe to use int(100) to set
286 // a uint8_t port, but not int(-42) or int(300)
287 if constexpr(std::is_arithmetic_v<T>)
288 {
289 if(mismatching && isCastingSafe(previous_type, value))
290 {
291 mismatching = false;
292 }
293 }
294
295 if(mismatching)
296 {
297 debugMessage();
298
299 auto msg = StrCat("Blackboard::set(", key,
300 "): once declared, "
301 "the type of a port shall not change. "
302 "Previously declared type [",
303 BT::demangle(previous_type), "], current type [",
304 BT::demangle(typeid(T)), "]");
305 throw LogicError(msg);
306 }
307 }
308 // if doing set<BT::Any>, skip type check
309 if constexpr(std::is_same_v<Any, T>)
310 {
311 previous_any = new_value;
312 }
313 else
314 {
315 // copy only if the type is compatible
316 new_value.copyInto(previous_any);
317 }
318 entry.sequence_id++;
319 entry.stamp = std::chrono::steady_clock::now().time_since_epoch();
320 }
321}

338template <typename T>
339inline Expected<Timestamp> Blackboard::getStamped(const std::string& key, T& value) const
340{
341 if(auto entry = getEntry(key))
342 {
343 std::unique_lock lk(entry->entry_mutex);
344 if(entry->value.empty())
345 {
346 return nonstd::make_unexpected(StrCat("Blackboard::getStamped() error. Entry [",
347 key, "] hasn't been initialized, yet"));
348 }
349 value = entry->value.cast<T>();
350 return Timestamp{ entry->sequence_id, entry->stamp };
351 }
352 return nonstd::make_unexpected(
353 StrCat("Blackboard::getStamped() error. Missing key [", key, "]"));
354}
